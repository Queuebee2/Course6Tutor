library(edgeR)
# reading in file
RNA.Seq <- read.delim("C:/Users/sanne/Downloads/RNA-Seq.txt", comment.char="#")

# adjusting file
RNA.Seq <- RNA.Seq[1:5]

# maakt twee groepen dus glucose en ribose
group <- c(1,1,2,2)

# maakt een bewerkbare lijst van de dataset
y <- DGEList(counts=RNA.Seq[2:5], group=group, genes = RNA.Seq[1])

# filtert genen die bijna niet tot expressie komen eruit
keep <- filterByExpr(y)

# hou alleen degene die dus bruikbaar zijn
y <- y[keep, , keep.lib.sizes=FALSE]

# normalizatie
y <- calcNormFactors(y)

# maakt een plot (The function plotMDS produces a plot in which distances between samples
#correspond to leading biological coefficient of variation (BCV) between those samples)
plotMDS(y)

# maakt de data voor dispersie plot  maar er moet nog een design bij  maar snap niet hoe ik dat moet maken dus deze klopt wss niet
disper <- estimateDisp(y)
## Design matrix not provided. Switch to the classic mode.
common <- estimateGLMCommonDisp(y)
trend <- estimateGLMTrendedDisp(y)
tag <- estimateGLMTagwiseDisp(common)


# plot of dispersion
plotBCV(disper)

plotBCV(common)

plotBCV(trend)

plotBCV(tag)

# plot mean variance
plotMeanVar(disper)
plotMeanVar(common)

plotMeanVar(trend)
plotMeanVar(tag)

# DE genes
glf <- glmFit(disper)
qlf <- glmLRT(glf)
topTags(qlf)
